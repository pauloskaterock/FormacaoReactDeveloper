useState

Gerencia estado em componentes funcionais.

const [state, setState] = useState(initialValue);

useEffect

Executa efeitos colaterais em componentes funcionais.

useEffect(() => {
  // Efeito aqui
  return () => {
    // Limpeza (opcional)
  };
}, [dependencies]);

useContext

Acessa o contexto sem precisar de props drilling.

const value = useContext(MyContext);

useReducer

const [state, dispatch] = useReducer(reducer, initialArg, init);

useRef

Cria referências mutáveis que persistem entre re-renderizações.

const refContainer = useRef(initialValue);

useMemo

Memoriza valores calculados para otimização.

const memoizedValue = useMemo(() => computeExpensiveValue(a, b), [a, b]);

useCallback

Memoriza funções para evitar recriação desnecessária.

const memoizedCallback = useCallback(() => {
  doSomething(a, b);
}, [a, b]);

useLayoutEffect

Similar ao useEffect, mas executa sincronamente após todas as mutações do DOM.

  // Código aqui
}, [dependencies]);

Personaliza a instância que é exposta quando usando ref em componentes.

useImperativeHandle(ref, () => {
  return {
    focus() {
      inputRef.current.focus();
    }
  };
}, []);
